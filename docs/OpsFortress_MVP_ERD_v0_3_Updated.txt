Table customer_accounts {
  id uuid [pk]
  account_name text [not null]
  account_status text [not null]
  billing_status text
  created_at timestamptz
  updated_at timestamptz
}

Table countries {
  id uuid [pk]
  country_name text [unique, not null]
  country_code text
  active_status boolean
}

Table business_identifier_types {
  id uuid [pk]
  country_id uuid [not null]
  identifier_label text [not null] // ABN, NZBN, Company Number, EIN, etc.
  identifier_code text
  active_status boolean
}

Table business_entities {
  id uuid [pk]
  legal_name text [not null]
  trading_name text
  country_id uuid [not null]
  entity_type text [not null]
  entity_type_other_text text
  blockchain_id uuid [unique, not null]
  status text
  review_status text
  created_by_user_id uuid
  created_at timestamptz
  updated_at timestamptz
}

Table business_identifiers {
  id uuid [pk]
  business_entity_id uuid [not null]
  identifier_type_id uuid [not null]
  identifier_value text [not null]
  normalised_identifier_value text [not null]
  is_primary boolean
  verification_status text
  created_at timestamptz
}

Table account_businesses {
  id uuid [pk]
  account_id uuid [not null]
  business_entity_id uuid [not null]
  relationship_type text // owned / controlled
  is_primary boolean
  status text
}

Table workplace_environments {
  id uuid [pk]
  environment_name text [not null] // Construction, Federal, Mine, Petroleum, Other
  active_status boolean
}

Table workplaces {
  id uuid [pk]
  account_id uuid [not null]
  business_entity_id uuid [not null]
  country_id uuid [not null]
  environment_id uuid
  workplace_name text [not null]
  address_line_1 text
  suburb text
  region text
  postcode text
  status text
}

Table industries {
  id uuid [pk]
  external_industry_id text [unique]
  industry_group text
  industry_sub_group text
  industry_leaf text
  industry_candidate_key text
  active_status boolean
}

Table business_industries {
  id uuid [pk]
  business_entity_id uuid [not null]
  industry_id uuid [not null]
  is_primary boolean
  created_at timestamptz
}

  id uuid [pk]

  name text [not null]
  email text [unique, not null]
  role text [not null]

  status text
}

  id uuid [pk]

  id uuid [pk]
  host_account_id uuid [not null]

  id uuid [pk]

  id uuid [pk]
  session_id uuid [not null]

  comment text
  evidence_file_id uuid

  id uuid [pk]
